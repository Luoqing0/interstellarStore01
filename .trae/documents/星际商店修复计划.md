# 星际商店 Mod 修复计划

## 问题总结

### 1. 报错问题
- **报错10 (主界面)**: `StarStore_TradeConditionDef is not a Def type` - 自定义 Def 类型未在 XML 中声明
- **报错11 (购买和出售)**: `Failed to generate trade drop center` + 物品放置失败 - 交易信标缺失导致降落点无效

### 2. UI 问题（长期存在）
- 商店界面和模组设置界面的行/列调整大概率失效，刷新界面时才可能有效
- 物品格过小时无法显示数量调整按钮和输入框
- 购物车界面商品名位置过左导致超出边框显示为"……"
- 购买的物品会出现在"出售"里，反之亦然
- 底部翻页按钮和物品格重合了

---

## 修复方案

### 阶段 1: 修复 Def 类型加载错误

**问题**: `StarStore_TradeConditionDef` 未在 XML 中声明为 Def 类型

**修复**: 创建 `Defs/DefTypes/StarStore_DefOf.xml`
```xml
<?xml version="1.0" encoding="utf-8"?>
<Defs>
  <!-- 声明自定义 Def 类型，确保 RimWorld 能正确识别 -->
  <StarStore_TradeConditionDef>
    <defName>StarStore_DefaultConditions</defName>
    <label>星际商店默认交易条件</label>
    <description>星际商店物品交易前置条件配置</description>
    <conditions>
      <!-- 示例：高级零部件需要研究"标准机械科技" -->
      <li>
        <thingDef>ComponentSpacer</thingDef>
        <requiredResearch>StandardMechanoid</requiredResearch>
      </li>
      <!-- 示例：电荷步枪需要研究"电荷武器" -->
      <li>
        <thingDef>Gun_ChargeRifle</thingDef>
        <requiredResearch>ChargeWeapons</requiredResearch>
      </li>
    </conditions>
  </StarStore_TradeConditionDef>
</Defs>
```

**同时修改**: 删除 `Defs/TradeConditionDefs/星际商店交易条件.xml` 中的重复定义

### 阶段 2: 修复交易降落点问题

**问题**: `DropCellFinder.TradeDropSpot` 返回无效位置 `(-1000, -1000, -1000)`

**修复**: 修改 `MainTabWindow_星际商店_交易.cs` 中的 `获取有效降落点` 方法

```csharp
private IntVec3 获取有效降落点(Map map)
{
    if (map == null) return IntVec3.Invalid;
    
    // 先尝试获取交易信标位置
    IntVec3 spot = DropCellFinder.TradeDropSpot(map);
    
    // 检查位置是否有效
    if (!spot.IsValid || spot.x < 0 || spot.z < 0)
    {
        // 回退到地图中心附近
        spot = map.Center;
        // 确保位置可通行
        if (!spot.Walkable(map))
        {
            // 搜索附近可通行位置
            spot = CellFinder.RandomClosewalkCellNear(spot, map, 10);
        }
    }
    
    return spot;
}
```

### 阶段 4: 修复物品格过小时数量控制不显示

```csharp
private void 绘制单个物品格子(Rect rect, ThingDef def, float 格子尺寸)

        // ... 数量控制代码 ...
    }

// 新增方法：迷你数量控制（只显示 +/- 按钮）
private void 绘制迷你数量控制(Rect rect, TransactionKey key)
{
    float 按钮宽 = rect.width / 2f - 2f;
    
    Rect 减按钮 = new Rect(rect.x, rect.y, 按钮宽, rect.height);
    if (Widgets.ButtonText(减按钮, "-"))
    {
        if (当前交易数量.ContainsKey(key) && 当前交易数量[key] > 0)
            当前交易数量[key]--;
    }
    
    Rect 加按钮 = new Rect(rect.x + 按钮宽 + 4f, rect.y, 按钮宽, rect.height);
    if (Widgets.ButtonText(加按钮, "+"))
    {
        if (!当前交易数量.ContainsKey(key))
            当前交易数量[key] = 0;
        当前交易数量[key]++;
    }
}
```

### 阶段 6: 修复购物车购买/出售物品混淆问题

**问题**: 购买的物品出现在"出售"列表，反之亦然

**修复**: 修改 `MainTabWindow_星际商店_购物车.cs`

```csharp
private void 绘制购物车(Rect rect)
{
    // ... 现有代码 ...
    
    // 分两个区域：购买清单 和 出售清单
    float y = rect.y + 32f;
    float halfHeight = (rect.yMax - y - 5f) / 2f;
    
    // === 上半部分：购买清单 ===
    Rect 购买区域 = new Rect(rect.x, y, rect.width, halfHeight);
    // 只显示购买交易数量中的物品
    绘制购物车分区(购买区域, true, "🛒 购买清单", 购买按钮色, ref 购物车购买滚动, 购买交易数量);
    
    // === 下半部分：出售清单 ===
    Rect 出售区域 = new Rect(rect.x, y + halfHeight + 2f, rect.width, halfHeight - 2f);
    // 只显示出售交易数量中的物品
    绘制购物车分区(出售区域, false, "💰 出售清单", 出售按钮色, ref 购物车出售滚动, 出售交易数量);
}

private void 绘制购物车分区(Rect rect, bool 购买模式分区, string 标题, Color 标题色, 
    ref Vector2 scrollPos, Dictionary<TransactionKey, int> 指定字典)
{
    // ... 标题绘制 ...
    
    // 使用传入的字典，而不是根据模式选择
    List<KeyValuePair<TransactionKey, int>> 分区物品 = new List<KeyValuePair<TransactionKey, int>>();
    float 分区总价 = 0f;
    
    foreach (var kv in 指定字典)  // 使用传入的指定字典
    {
        if (kv.Value > 0)
        {
            float 单价 = 购买模式分区 
                ? 获取购买价格(kv.Key.def, kv.Key.quality, kv.Key.stuff)
                : 获取出售价格(kv.Key.def, kv.Key.quality, kv.Key.stuff);
            分区物品.Add(kv);
            分区总价 += 单价 * kv.Value;
        }
    }
    
    // ... 其余代码不变 ...
}
```

更好的修复方案是调整主窗口布局，为分页预留空间：

```csharp
// 在 MainTabWindow_星际商店.cs 的 DoWindowContents 中
public override void DoWindowContents(Rect inRect)
{
    // ... 渐变背景 ...
    
    // ===== 顶部标题栏 =====
    Rect 标题区域 = new Rect(inRect.x + 水平内边距, inRect.y + 标题栏Y偏移, inRect.width - 水平内边距 * 2f, 标题栏高);
    绘制标题栏(标题区域);
    
    // ===== 分类标签页 =====
    Rect 分类区域 = new Rect(inRect.x + 水平内边距, 标题区域.yMax + 区域间距, inRect.width - 水平内边距 * 2f, 分类标签栏高);
    绘制分类标签页(分类区域);
    
    // ===== 物品网格区域（预留分页空间）=====
    float 分页栏高 = 30f;  // 新增：为分页预留高度
    float 网格X = inRect.x + 水平内边距;
    float 网格Y = 分类区域.yMax + 4f;
    float 当前购物车宽 = 显示购物车 ? 购物车宽 : 0f;
    float 网格宽 = inRect.width - 水平内边距 * 2f - 当前购物车宽 - (显示购物车 ? 购物车间距 : 0f);
    float 网格高 = inRect.yMax - 4f - 网格Y - 网格底部留白 - 分页栏高;  // 减去分页栏高度
    
    Rect 网格区域 = new Rect(网格X, 网格Y, 网格宽, 网格高);
    绘制物品网格(网格区域);
    
    // ===== 分页控件（在网格下方）=====
    Rect 分页区域 = new Rect(网格X, 网格区域.yMax + 2f, 网格宽, 分页栏高);
    绘制分页控件(分页区域);
    
    // ... 其余代码 ...
}

// 新增方法：绘制分页控件
private void 绘制分页控件(Rect rect)
{
    int 总页数 = Mathf.Max(1, Mathf.CeilToInt((float)当前显示物品.Count / 每页物品数));
    当前页码 = Mathf.Clamp(当前页码, 1, 总页数);
    
    if (总页数 <= 1) return;
    
    float 按钮宽 = 24f;
    float 分页Y = rect.y + (rect.height - 18f) / 2f;  // 垂直居中
    
    // 首页按钮
    Rect 首页Rect = new Rect(rect.x + 4f, 分页Y, 按钮宽, 18f);
    if (Widgets.ButtonText(首页Rect, "|<")) { 当前页码 = 1; }
    
    // 上一页按钮
    Rect 上一页Rect = new Rect(rect.x + 30f, 分页Y, 按钮宽, 18f);
    if (Widgets.ButtonText(上一页Rect, "<")) { if (当前页码 > 1) 当前页码--; }
    
    // 页码显示
    GUI.color = 文字色;
    Text.Font = GameFont.Tiny;
    Widgets.Label(new Rect(rect.x + 60f, 分页Y + 2f, 80f, 16f),
        "StarStore_PageInfo".Translate(当前页码, 总页数));
    Text.Font = GameFont.Small;
    GUI.color = Color.white;
    
    // 下一页按钮
    Rect 下一页Rect = new Rect(rect.x + 140f, 分页Y, 按钮宽, 18f);
    if (Widgets.ButtonText(下一页Rect, ">")) { if (当前页码 < 总页数) 当前页码++; }
    
    // 末页按钮
    Rect 末页Rect = new Rect(rect.x + 166f, 分页Y, 按钮宽, 18f);
    if (Widgets.ButtonText(末页Rect, ">|")) { 当前页码 = 总页数; }
    
    // 跳转输入框
    Rect 跳转Rect = new Rect(rect.x + 200f, 分页Y, 40f, 18f);
    Widgets.TextFieldNumeric(跳转Rect, ref 当前页码, ref 跳转页码输入, 1, 总页数);
}
```
